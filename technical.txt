
### **Advanced Search Query**
```sql
-- Example search with full-text search and filters
SELECT DISTINCT a.*,
       ts_rank(search_vector, plainto_tsquery('english', %s)) as rank,
       array_agg(DISTINCT t.name) as topic_names,
       array_agg(DISTINCT tag.name) as tag_names
FROM articles a
LEFT JOIN article_topics at ON a.id = at.article_id
LEFT JOIN topics t ON at.topic_id = t.id
LEFT JOIN article_tags atag ON a.id = atag.article_id
LEFT JOIN tags tag ON atag.tag_id = tag.id
WHERE a.status = 'published' 
  AND a.access_level <= %s
  AND (search_vector @@ plainto_tsquery('english', %s) OR %s = '')
  AND (article_type_id = ANY(%s) OR %s = '{}')
ORDER BY rank DESC, a.created_at DESC
LIMIT %s OFFSET %s;
```

```python
class DownloadManager:
    async def check_download_permission(self, user_id):
        count = await get_user_download_count(user_id)
        user = await get_user(user_id)
```
